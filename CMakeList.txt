cmake_minimum_required(VERSION 3.5)
project(image_mask_publisher)

# Set the C++ standard (adjust as necessary)
if(NOT CMAKE_CXX_STANDARD)
  set(CMAKE_CXX_STANDARD 14)
endif()

# Find dependencies
find_package(ament_cmake REQUIRED)
find_package(rclcpp REQUIRED)
find_package(sensor_msgs REQUIRED)
find_package(std_msgs REQUIRED)
find_package(OpenCV REQUIRED)
find_package(cv_bridge REQUIRED)
find_package(custom_msgs REQUIRED)  # Your custom message package

# Add the executable for the node.
add_executable(image_mask_publisher_node src/image_mask_publisher.cpp)

# Link dependencies.
ament_target_dependencies(image_mask_publisher_node
  rclcpp
  sensor_msgs
  std_msgs
  OpenCV
  cv_bridge
  custom_msgs
)

# Install the executable
install(TARGETS
  image_mask_publisher_node
  DESTINATION lib/${PROJECT_NAME}
)

ament_package()
